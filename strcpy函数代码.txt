#include<stdio.h>
#include<assert.h>
char* my_strcpy(char * dest, const char * src)
{
	char* ret = dest;
	assert(src != NULL);//断言
	assert(dest != NULL);
	while (*src != '\0')
	{
		*dest++ = *src++;
	}
	return ret;
}
int main()
{
	char arr[20] = "abcdef";
	char *c2 = "hello world!";
	printf("c1=%s\n c2=%s\n", arr, c2);
	my_strcpy(arr, c2);
	printf("%s\n", my_strcpy(arr, c2));
	return 0;
}
